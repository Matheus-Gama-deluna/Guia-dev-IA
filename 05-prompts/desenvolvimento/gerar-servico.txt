## Resposta Esperada (Exemplo - NestJS)

```typescript
@Injectable()
export class AgendamentoService {
  constructor(
    private readonly agendamentoRepository: AgendamentoRepository,
    private readonly notificacaoService: NotificacaoService,
  ) {}

  async criar(dto: CriarAgendamentoDto): Promise<Agendamento> {
    // Validar horário disponível
    const conflito = await this.agendamentoRepository.findByHorario(dto.dataHora);
    if (conflito) {
      throw new ConflictException('Horário já ocupado');
    }

    // Criar agendamento
    const agendamento = await this.agendamentoRepository.create({
      ...dto,
      status: StatusAgendamento.PENDENTE,
    });

    // Notificar cliente
    await this.notificacaoService.enviarConfirmacao(agendamento);

    return agendamento;
  }
}
```
